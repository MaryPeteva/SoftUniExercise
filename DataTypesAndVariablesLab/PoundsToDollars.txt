using System;

namespace PoundsToDollars
{
    class Program
    {
        static void Main(string[] args)
        {
            decimal pound = decimal.Parse(Console.ReadLine());
            decimal usd = (pound * 131)/100;
            //1 British Pound = 1.31 Dollars
            Console.WriteLine($"{usd:F3}");
        }
    }
}
