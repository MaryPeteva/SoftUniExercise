using System;

namespace NxNMatrix
{
    class Program
    {
        static void Main(string[] args)
        {
            int num = int.Parse(Console.ReadLine());

            PrintRow(num);
        }

        private static void PrintRow(int num)
        {
            for (int i = 0; i < num; i++)
            {
                PrintLine(num);
                Console.WriteLine();
            }
        }

        private static void PrintLine(int num)
        {
            for (int i = 0; i < num; i++)
            {
                Console.Write($"{num} ");
            }

          
        }
    }
}
