using System;

namespace ReverseString
{
    class Program
    {
        static void Main(string[] args)
        {
            string input = Console.ReadLine();
            string mirror = "";
            for (int i = input.Length-1; i>=0; i--)
            {
                char symb = input[i];
                mirror += symb;
            }
            Console.WriteLine(mirror);
        }
    }
}
